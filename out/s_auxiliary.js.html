orderByArray = function (docs,arr){
    for (var i = 0 ; i < arr.length ; i++){
       for (var j = i ; j < arr.length ; j++){
            if (docs[j].sessionId == arr[i]){
                var doc = docs[i];
                docs[i] = docs[j];
                docs[j]=doc;
                continue;
            }
        } 
    }
    return docs;
}

createKeyValJSON=function  (arr , key){
    var temp = {}, uid = '' ,count=0;
    for ( k in arr ){
        if (count==4 && uid == arr[k][key]) continue;
        if (uid != arr[k][key])
        {
            count = 0;
            uid = arr[k][key];
            temp[uid] = [];
        }
        count++;
        temp[uid].push(arr[k]);
    }
    return temp;
}
